<task type="auto">
  <name>Task 5: Frontend API client for projects</name>
  <files>frontend/src/api/projects.ts, frontend/src/types/project.ts</files>
  <action>
1. `types/project.ts`:
   ```typescript
   export interface Project {
     id: string; name: string; key: string; description: string;
     is_archived: boolean; created_at: string; updated_at: string;
     member_count: number; active_sprint_name: string | null;
   }
   export interface ProjectDetail extends Project {
     members: Member[]; columns: Column[];
   }
   export interface Member {
     id: string; user: User; role: 'owner' | 'admin' | 'developer' | 'viewer'; joined_at: string;
   }
   ```

2. `api/projects.ts`:
   ```typescript
   export const projectsApi = {
     list: (archived?: boolean) => client.get<Project[]>('/projects/', { params: { is_archived: archived } }),
     create: (data: { name: string; description?: string }) => client.post<Project>('/projects/', data),
     get: (id: string) => client.get<ProjectDetail>(`/projects/${id}/`),
     update: (id: string, data: Partial<Project>) => client.put<Project>(`/projects/${id}/`, data),
     archive: (id: string) => client.delete(`/projects/${id}/`),
     invite: (id: string, data: { email: string; role?: string }) => client.post(`/projects/${id}/invite/`, data),
     getMembers: (id: string) => client.get<Member[]>(`/projects/${id}/members/`),
     updateMemberRole: (projectId: string, userId: string, role: string) =>
       client.put(`/projects/${projectId}/members/${userId}/`, { role }),
   };
   ```
  </action>
